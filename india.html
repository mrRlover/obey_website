  mutate(across(where(is.numeric), function(x) c(NA,diff(log(x))))) %>% 
  slice(-1)

I3 = in_index %>%
  mutate(DATE = ymd(Date), Close = as.numeric(Close)) %>% 
  right_join(in_temp) %>% 
  select(-c(Date, TEMP)) %>% 
  arrange(DATE) %>% 
  mutate(across(-DATE, na.locf)) %>% 
  mutate(across(.cols = where(is.numeric),
                function(x) ifelse(x == 0, NA, x))) %>%
  mutate(across(.cols = where(is.numeric), function(x) c(NA,diff(log(x))))) %>%
  slice(-1)

in_mret = in_ret %>% 
  select(-DATE) %>% 
  rowMeans(na.rm = T) %>% 
  as.data.frame() %>% 
  mutate(DATE = in_ret$DATE) %>% 
  select(DATE, everything()) %>% 
  `colnames<-`(c("Date", "Mean"))

3.2 Market cap weighted

Portfolio return using market cap as weights. Here, market cap is proxied by volume. The individual share weight is share volume divided by total volume

w = in_volume %>% 
  mutate(DATE = as.Date(Date, origin = "1899-12-31")) %>% 
  select(DATE, everything()) %>% 
  right_join(in_temp) %>% 
  arrange(DATE) %>% 
  select(-c(Date, TEMP)) %>%
  select(DATE, everything()) %>% 
  slice(-1)

w[, -1] = w[, -1] / rowSums(w[, -1], na.rm = T)

The portfolio return is the average of the return multiplied by the weights.

in_mret_cap = rowMeans(in_ret[, -1] * w[, -1], na.rm = T) %>% 
  as.data.frame() %>% 
  mutate(DATE = in_ret$DATE) %>% 
  select(DATE, everything()) %>% 
  `colnames<-`(c("Date", "Mean"))
