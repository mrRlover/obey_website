sa_ret = sa_price %>%
  mutate(across(where(is.numeric), function(x) ifelse(x==0, NA, x))) %>% 
  mutate(across(where(is.numeric), function(x) c(NA,diff(log(x))))) %>% 
  slice(-1)

jse = sa_index %>%
  mutate(DATE = ymd(Date)) %>% 
  right_join(sa_temp) %>% 
  select(-c(Date, TEMP)) %>% 
  arrange(DATE) %>% 
  mutate(across(-DATE, function(x) c(NA, na.locf(x)))) %>% 
  mutate(across(.cols = where(is.numeric),
                function(x) ifelse(x == 0, NA, x))) %>%
  mutate(across(.cols = where(is.numeric), function(x) c(NA,diff(log(x))))) %>%
  slice(-1)

sa_mret = sa_ret %>% 
  select(-DATE) %>% 
  rowMeans(na.rm = T) %>% 
  as.data.frame() %>% 
  mutate(DATE = sa_ret$DATE) %>% 
  select(DATE, everything()) %>% 
  `colnames<-`(c("Date", "Mean"))

w = sa_volume %>%
