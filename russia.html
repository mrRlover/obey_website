ru_policy = ru_policy %>% 
  mutate(DATE = ymd(Week)) %>% 
  right_join(ru_temp) %>% 
  arrange(DATE) %>%
  select(-Week) %>% 
  mutate(Week = paste0(year(DATE), week(DATE))) %>% 
  group_by(Week) %>% 
  mutate(climate_change = ifelse(n() < 6, `climate change: (Russia)`,
        na.locf(`climate change: (Russia)`, fromLast = T))) %>% 
  ungroup() %>% 
  mutate(POLICY = c(NA, diff(log(climate_change)))) %>% 
  mutate(POLICY = case_when(is.na(POLICY) ~ 0, is.nan(POLICY) ~ 0,
                            is.infinite(POLICY) ~ 0,
                            TRUE ~ POLICY))

R3 = ru_index %>%
  mutate(DATE = ymd(Date), `Adj Close` = as.numeric(`Adj Close`)) %>% 
  right_join(ru_temp) %>% 
  select(-c(Date, TEMP)) %>% 
  arrange(DATE) %>% 
  mutate(across(-DATE, na.locf)) %>% 
  mutate(across(.cols = where(is.numeric),
                function(x) ifelse(x == 0, NA, x))) %>%
  mutate(across(.cols = where(is.numeric), function(x) c(NA,diff(log(x))))) %>%
  slice(-1)

ru_mret = ru_ret %>%
